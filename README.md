# GitProject

# Создание репозитория
  # mkdir - создать каталог
  # cd - перейти в каталог
  # git init - создание репозитория
  # git add - добавление файла в индекс
  # git commit -m "..." - коммит нового файла
  # git status - проверка состояния репозитория
  # 
# Управление файлами репозитория
  # echo 'text' > file.txt - запись в файл
  # git commit -am 'name' -запись изменения в историю
  # git reset - сброс изменения
  # cat file - просмотр файла
  # touch - создать файл
  # git rm - удалить файл
  # git mv - перемещение файла
  # ls - список файлов
# История изменения
  # git log - история изменений
  # git log --oneline - история изменений в одну строку
  # git revert - возврат к предыдущему коммиту
  # git reflog - подробная история
  # git log --all - история всех веток
  # git reset --hard - возврат к начальной точке на ветке
  #
# Ветки репозитория
  # git branch - просмотр созданных веток
  # git branch name - создание ветки
  # git branch -d name - удаление ветки
  # git checkout - отмена изменений на ветке
  # git checkout name - переход к другой ветке
  # git checkout -b name - создание и переход к новой ветке
  # 
# Публикация репозитория
  # git clone - клонирование репозитория
  # git remote - удаленные репозитории
  # git fetch- синхронизация репозиториев
  # git pull - импорт изменений
  # git push - экспорт изменений
  # 
  # 
# Слияние веток
  # git merge - слияние веток
  # git rebase - перебазирование веток
  # 
  # 
  # 
  # 
  # 
# Управление версиями
  # git tag - задать версию 
  # git tag -d - удалить версию
  # git push origin --tags - загрузить версию в интернет
  # git push origin --delete - удалить версию из интернета
  # 
  # 
  # 
# Создание pull request
  # 
# Сложные операции
  # git commit --amend -m - внесение изменнеий в коммит без создания нового коммита
  # git cherry-pick - добавление коммита с другой ветки
  # git stash - сохранение редактирования во временном хранилище
  # git squash - обьединение коммитов
